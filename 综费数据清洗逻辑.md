### 1.1 OA报销数据（tztek_t_personcost_bx）清洗

```sql
-- 清洗1: 删除审核日期为空的记录（未审核的报销不应计入综费）
DELETE FROM tztek_t_personcost_bx
WHERE FIDATE IS NULL OR FIDATE = ''

WHERE FINTYPE IS NULL OR FINTYPE = ''

-- 清洗3: 删除事业部为空的记录（无法归属）
DELETE FROM tztek_t_personcost_bx
WHERE SYBMC IS NULL OR SYBMC = ''

-- 清洗4: 修正金额格式（去除非数字字符）
UPDATE tztek_t_personcost_bx
SET ACTUALAMOUNT = REPLACE(REPLACE(ACTUALAMOUNT, ',', ''), ' ', '')
WHERE ISNUMERIC(REPLACE(REPLACE(ACTUALAMOUNT, ',', ''), ' ', '')) = 1
  AND ISNUMERIC(ACTUALAMOUNT) = 0

-- 清洗5: 删除金额无法转换的记录
DELETE FROM tztek_t_personcost_bx
WHERE ISNUMERIC(ACTUALAMOUNT) = 0

-- 清洗6: 修正提交日期格式
UPDATE tztek_t_personcost_bx
SET TJRQ = CONVERT(NCHAR(10), CAST(TJRQ AS DATE), 120)
WHERE ISDATE(TJRQ) = 1 AND LEN(TJRQ) <> 10
```

### 2.2 部门补全

```sql
-- 清洗1: 按入职代码+年月关联补全部门（取每月最后一条）
UPDATE A
SET A.部门 = B.SSBM
FROM TZTEK_T_COSTDETL A
INNER JOIN (
    SELECT RZDM, LEFT(TIME, 7) AS YM, SSBM,
           ROW_NUMBER() OVER (PARTITION BY RZDM, LEFT(TIME, 7) ORDER BY TIME DESC) AS RN
    FROM TZTEK_T_BUSMODULE
    WHERE YEAR(CAST(TIME AS DATE)) = YEAR(GETDATE())
) B ON A.预定人入职代码 = B.RZDM
    AND LEFT(A.流程提交日期, 7) = B.YM
    AND B.RN = 1
WHERE (A.部门 IS NULL OR A.部门 = '')

-- 清洗2: 仍未匹配的，取该入职代码最近一条记录
UPDATE A
SET A.部门 = B.SSBM
FROM TZTEK_T_COSTDETL A
INNER JOIN (
    SELECT RZDM, SSBM,
           ROW_NUMBER() OVER (PARTITION BY RZDM ORDER BY TIME DESC) AS RN
    FROM TZTEK_T_BUSMODULE
) B ON A.预定人入职代码 = B.RZDM AND B.RN = 1
WHERE (A.部门 IS NULL OR A.部门 = '')

-- 清洗3: 仍未匹配的，标记为'未知部门'
UPDATE TZTEK_T_COSTDETL
SET 部门 = '未知部门'
WHERE 部门 IS NULL OR 部门 = ''
```

### 2.3 项目号匹配

```sql
-- 清洗1: 项目号非空且非'public'时，关联项目主数据更新XMH
UPDATE A
SET A.XMH = B.XMH,
    A.所属业务中心 = B.SYBMC,
    A.事业部 = B.XMDYBMMC,
    A.项目名称 = B.XMMC
FROM TZTEK_T_COSTDETL A
INNER JOIN (
    SELECT XMH, XMMC, SYBMC, XMDYBMMC
    FROM TZTEK_V_LXXMHXX_ALL
    GROUP BY XMH, XMMC, SYBMC, XMDYBMMC
) B ON A.项目号 LIKE B.XMH + '%'
WHERE A.XMH NOT IN ('', 'public', '技术平台')
  AND ISNULL(A.事业部, '') <> '矽行'

-- 清洗2: 矽行事业部特殊处理（跳过项目号匹配，保留原始事业部）
-- 无需操作，矽行数据保持原样
```

### 2.4 事业部名称规范化

```sql
-- 清洗1: 事业部名称标准化
UPDATE TZTEK_T_COSTDETL
SET 事业部 = CASE
    WHEN 事业部 = '大客户事业部' THEN '大客户其它'
    WHEN 事业部 = '半导体事业部' THEN '半导体其它'
    WHEN 事业部 = '光电事业部'   THEN '光电其它'
    WHEN 事业部 = '智能事业部'   THEN '智能其它'
    WHEN 事业部 = '机器人事业部' THEN '机器人其它'
    WHEN 事业部 = '计量事业部'   THEN '计量其它'
    WHEN 事业部 = 'MueTec子公司' THEN 'MueTec其它'
    WHEN 事业部 = '其它'         THEN '运营中心'
    ELSE 事业部
END
WHERE 事业部 IN ('大客户事业部','半导体事业部','光电事业部',
                  '智能事业部','机器人事业部','计量事业部',
                  'MueTec子公司','其它')

-- 清洗2: 事业部含"事业部"但不在上述列表中的，替换为"XX其它"
UPDATE TZTEK_T_COSTDETL
SET 事业部 = REPLACE(事业部, '事业部', '其它')
WHERE 事业部 LIKE '%事业部' 
  AND 事业部 NOT IN ('大客户其它','半导体其它','光电其它',
                      '智能其它','机器人其它','计量其它','MueTec其它')
  AND 事业部 NOT LIKE '%中心%'
```

### 2.5 矽行系数打折

```sql
-- 清洗1: 获取矽行系数
DECLARE @xixing_rate FLOAT
SELECT @xixing_rate = CAST(KEYWPRD_NUM AS FLOAT)
FROM TZTEK_T_KEYWORD
WHERE KEYWORD_NAME = '矽行系数'
  AND YEAR(CRE_DT) = YEAR(GETDATE())

-- 清洗2: 对矽行费用打折（排除EBS/行政采购/资产处置报废）
UPDATE TZTEK_T_COSTDETL
SET 结算金额 = 结算金额 * @xixing_rate
WHERE 事业部 = '矽行'
  AND 分类 NOT IN ('其它_EBS付款','EBS_行政采购申请','报废_资产处置报废')
```

## 八、投标保证金数据清洗

```sql
-- 清洗1: 删除2026年之前的数据（统计起始2026年）
DELETE FROM TZTEK_T_COSTDETL
WHERE 科目 = '竞标/投标保证金'
  AND LEFT(账期, 4) < '2026'

-- 清洗2: 验证跨年冲减逻辑
-- 归还金额的负数冲减不应超过原始支付金额
SELECT A.订单号, A.支付金额, B.冲减总额
FROM (
    SELECT 订单号, SUM(结算金额) AS 支付金额
    FROM TZTEK_T_COSTDETL
    WHERE 科目 = '竞标/投标保证金' AND 结算金额 > 0
    GROUP BY 订单号
) A
LEFT JOIN (
    SELECT 订单号, ABS(SUM(结算金额)) AS 冲减总额
    FROM TZTEK_T_COSTDETL
    WHERE 科目 = '竞标/投标保证金' AND 结算金额 < 0
    GROUP BY 订单号
) B ON A.订单号 = B.订单号
WHERE B.冲减总额 > A.支付金额
-- 期望结果: 0行（冲减不应超过支付）
```

## 九、C19H457项目拆分清洗（仅2024年）

```sql
-- 清洗1: 按拆分比例重新分配C19H457费用
-- 先备份原始数据
SELECT * INTO #C19H457_backup
FROM TZTEK_T_COSTDETL
WHERE XMH = 'C19H457' AND 年度 = '2024'

-- 删除原始C19H457数据
DELETE FROM TZTEK_T_COSTDETL
WHERE XMH = 'C19H457' AND 年度 = '2024'

-- 按拆分比例插入子项目数据
INSERT INTO TZTEK_T_COSTDETL (...)
SELECT 
    B.子项目号 AS XMH,
    A.年度, A.账期, A.费用类别, A.科目, A.流程提交日期, A.财务审核日期,
    A.预定人入职代码, A.预订人, A.人员类别, A.所属中心, A.部门,
    B.子项目号 AS 项目号,
    C.XMMC AS 项目名称,
    A.订单号,
    A.结算金额 * B.比例 AS 结算金额,
    A.费用用途, A.所属业务中心, A.事业部, A.分类
FROM #C19H457_backup A
INNER JOIN TZTEK_T_ORDER_JiLiangSyb_XMH B ON 1=1
LEFT JOIN TZTEK_V_LXXMHXX_ALL C ON B.子项目号 = C.XMH
```

## 十、全量数据清洗执行顺序

```
Step 1:  源头数据清洗（1.1~1.4）
Step 2:  执行TZTEK_P_C_OACOST主存储过程（含子SP）
Step 3:  COSTDETL空值处理（2.1）
Step 4:  COSTDETL部门补全（2.2）
Step 5:  COSTDETL项目号匹配（2.3）
Step 6:  COSTDETL事业部名称规范化（2.4）
Step 7:  COSTDETL矽行系数打折（2.5）
Step 8:  COSTDETL排除规则清洗（2.6）
Step 9:  特别成本去重与标记（三）
Step 10: 现金流清洗（四）
Step 11: 累计采购清洗（五）
Step 12: 水电煤清洗（六）
Step 13: MueTec验证（七）
Step 14: 投标保证金验证（八）
Step 15: C19H457拆分（九，仅2024年）
Step 16: 执行质量检验规则验证
```

> **注意：** 实际生产环境中，Step 2（执行主存储过程）已包含Step 3~Step 8的大部分逻辑。上述清洗步骤主要用于数据修复场景或存储过程执行后的补充清洗。
